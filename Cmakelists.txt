cmake_minimum_required(VERSION 3.16)
project(my_project)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)
FetchContent_Declare(
  googletest
  URL https://github.com/google/googletest/archive/03597a01ee50ed33e9dfd640b249b4be3799d395.zip
)
set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)

add_executable(my_project
  src/main.cpp
)

add_executable(myTests
    tests/main-test.cpp
)

target_link_libraries(myTests 
    GTest::gtest_main
)

target_include_directories(my_project PRIVATE ${CMAKE_SOURCE_DIR})
target_include_directories(myTests PRIVATE ${CMAKE_SOURCE_DIR})

include(GoogleTest)
gtest_discover_tests(myTests)

if(MSVC) 
  target_compile_options(my_project PRIVATE /W4)
else() 
  target_compile_options(my_project PRIVATE -Wall -Wextra -pedantic) 
endif()
